class Solution {
public:
    
    int sumDigits(int number)
    {
        int sum = 0;
        while(number)
        {
            sum += number%10;
            number /= 10;
        }
        return sum;
    }
    
    int differenceOfSum(vector<int>& nums) 
    {
        int eSum = 0, dSum = 0;
        for(auto const c : nums)
        {
            eSum += c;
            dSum += sumDigits(c);
        }
        return abs(eSum - dSum);
    }
};
