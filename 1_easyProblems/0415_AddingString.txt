class Solution {
public:
    string addStrings(string num1, string num2) 
    {
        int c = 0, d = 0;
        for(auto i : num1)
        {
            c *= 10;
            c += i - '0';
        }
        for(auto i : num2)
        {
            d *= 10;
            d += i - '0';
        }
        int ans = c + d;
        auto res = to_string(ans);
        return res;
    }
};
