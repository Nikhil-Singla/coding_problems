class Solution {
public:
    int arrayPairSum(vector<int>& nums) 
    {
        sort(nums.begin(), nums.end());
        int sum = 0; 
        uint counter = 0;
        for(uint i = 0; i < (nums.size()/2); i++, counter++)
        {
            sum += min(nums[counter], nums[++counter]);
        }    
        return sum;
    }
};
