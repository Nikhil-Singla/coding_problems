class Solution {
public:
    int maxPower(string s) 
    {
        uint c = 1, ans = 1;
        for(uint i = 1; i < s.length(); i++)
        {
            if(s[i] == s[i-1])
            {
                c++;
                if(c > ans) {ans = c;}
            }
            else c = 1;
        }    
        return ans;
    }
};
