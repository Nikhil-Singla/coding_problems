class Solution {
public:
    int maxPower(string s) 
    {
        uint count = 1, ans = 1;
        for(uint i = 1; i < s.length(); i++)
        {
            if(s[i] == s[i-1])
            {
                count++;
                if(count > ans)
                {
                    ans = count;
                }
            }
            else
            count = 0;
        }    
        return ans;
    }
};
