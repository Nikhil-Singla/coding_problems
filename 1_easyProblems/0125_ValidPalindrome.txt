#include <cctype>
class Solution {
public:
    bool isPalindrome(string s) 
    {
        string phrase;
        for(auto c : s)
        {
            if(isalpha(tolower(c)))
            {
                phrase += tolower(c);
            }
        }
        int start = 0, end = phrase.length()-1;
        while(start < end)
        {
            if(phrase[start++] != phrase[end--])
            {
                return false;
            }
        }
        return true;
    }
};
