class Solution {
public:
    int transform(int num)

    }

    int getLucky(string s, int k) 
    {
        int sum = 0;
        k--;
        for(int i = 0; i < s.length(); i++)
        {
            char temp = s[i];
            int num_form = temp - 96; 
            while(num_form != 0)
            {
                sum *= 10;
                sum += num_form/10;
                num_form /= 10;
            }
        }

    }
};
