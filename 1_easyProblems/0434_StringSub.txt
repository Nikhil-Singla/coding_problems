class Solution {
public:
    int countSegments(string s) 
    {
        uint count = 0;
        bool flag = false;
        if(!s.length()) {return 0;}
        for(auto i : s)
        {
            if(i == ' ')
            {count++;}
            else
            {flag = true;}
        }    
        if(flag)
        return count+1;
        else
        return 0;
    }
};
