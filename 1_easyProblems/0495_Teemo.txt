class Solution {
public:
    int findPoisonedDuration(vector<int>& tS, int duration) 
    {
        int seconds = 0;
        for(int i = 0; i < tS[i].size() - 1; i++)
        {
            int time = tS[i] + duration;
            if(time > tS[i+1])
            {
                seconds += time - tS[i+1];
            }
            else
            {
                seconds += duration;
            }
        }    
    }
};
