class Solution {
public:
    int smallestEqual(vector<int>& nums) 
    {
        for(int i = 0, count = 0; i < nums.size(); i++, count++)
        {
            if(nums[i] == count)
            {
                return i;
            }
            if(count == 10)
            {
                count = 0;
            }
        }
        return -1;
    }
};
