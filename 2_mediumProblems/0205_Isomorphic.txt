class Solution {
public:
    bool isIsomorphic(string s, string t) 
    {
        unordered_map <char, char> map;
        for(int i = 0; i < s.length(); i++)
        {
            char key = s[i];
            if(map.find(key) == map.end())
            {
                map[key] = t[i];
            } 
            else
            {
                if(map[key] != t[i])
                {
                    return false;
                } 
            }
        }    
        return true;
    }
};
