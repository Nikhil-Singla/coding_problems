class Solution {
public:
    int maxIceCream(vector<int>& costs, int coins) 
    {
        uint i;
        if(coins < costs[0])
        {
            return 0;
        }
        for(i = 0; i < costs.size(); ++i)
        {
            if(coins > costs[i])
            {
                coins -= costs[i];
            }
            else
            {
                return i+1;
            }
        }
        return i;
    }
};
